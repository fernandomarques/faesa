temperaturas <- data.frame(celsius = c(28,26,26,24,24,26,32,33))


# Converte de Celsius para Kelvin
celsius_to_kelvin <- function(temp_C) {
	temp_K <- temp_C + 273.15
	return(temp_K)
}
celsius_to_kelvin(temperaturas$celsius)
temperaturas["Kelvin"] = celsius_to_kelvin(temperaturas$celsius)
temperaturas
lapply(temperaturas,mean)
