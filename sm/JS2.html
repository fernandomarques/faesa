String

  • Vimos que se somar um número com string, temos uma string!
  • Para saber o tamanho da string basta fazer str.length , não é um método!
  • str.indexOf('palavra') retorna a primeira posição da palavra em str
  • Também existe o str.lastIndexOf('palavra'), ambos retornam -1 se não for encontrado
  • str.slice(inicio, fim), também existe str.substring(inicio, fim). Slide aceita indice negativo!
  • str.substr(inicio, tamanho) também é parecido, mas o segundo parametro é o tamanho

String

  • str.replace("Oi","Tchau"), substitui todas ocorrências de Oi por Tchau
  • str.toUpperCase() e str.toLowerCase()
  • str.charAt(0) e str.charCodeAt(0)
  • Não devemos trabalhar com string como array. Ou seja, não podemos fazer str[0] = 'A'
  • Para converter em array podemos usar o str.split()
  • Para juntar podemos usar .toString() ou .join(sep) onde sep é o separador, por padrão ','

Objetos Protótipos

  • Todo objeto js tem um protótipo. O protótipo também é um objeto
  • 
function pessoa(nome, sobrenome, nickname){
	this.nome = nome;
	this.sobrenome = sobrenome;
	this.nickname = nickname;
}
pessoa.prototype.tipo = "Super-Herói";
pessoa.prototype.fullname = function(){
		return this.nome + ' ' + this.sobrenome;
}
var batman = new pessoa('Bruce','Wayne','Batman');
var luke = new pessoa('Luke','Cage', 'Powerman');

Objetos Protótipos-Herança

  • 
function Mammal(name){
	this.name=name;
	this.offspring=[];
}
Mammal.prototype.toString=function(){
	return '[Mammal "'+this.name+'"]';
}
Cat.prototype = new Mammal();        // Here's where the inheritance occurs
Cat.prototype.constructor=Cat;       // Otherwise instances of Cat would have a constructor of Mammal
function Cat(name){
	this.name=name;
}
Cat.prototype.toString=function(){
	return '[Cat "'+this.name+'"]';
}
var someAnimal = new Mammal('Mr. Biggles');
var myPet = new Cat('Felix');
alert('someAnimal is '+someAnimal);   // results in 'someAnimal is [Mammal "Mr. Biggles"]'
alert('myPet is '+myPet);             // results in 'myPet is [Cat "Felix"]'
