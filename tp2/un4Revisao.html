	char frase[30] = "String eh facil!";
	int i;
	for(i=0;i < strlen(frase); i++)
		printf("%c ",frase[i]);

Contar quantas vezes um caracter aparece em uma string


#include <stdio.h>
#include <string.h>
int strcontchr(char frase[], char chr){
	int i, cont =0;
	for(i=0 ; i < strlen(frase); i++){
		if(frase[i] == chr)
			cont++;
	}
	return cont;
}
int main(){
	char frase[30] = "String eh facil!";
	printf("%d ",strcontchr(frase,chr));
}

Contar letras maiúsculas e minúsculas

#include <stdio.h>
#include <string.h>
void strcontMaiMin(char frase[], int *cont_mai, int *cont_min){
	int i;
	*cont_mai = 0; *cont_min = 0;
	for(i=0 ; i < strlen(frase); i++){
		if(frase[i] >= 'A' && frase[i] <= 'Z')
			*cont_mai+= 1;
		else if(frase[i] >= 'a' && frase[i] <= 'z')
			*cont_min+= 1;
	}
}
int main(){
	char frase[30] = "String Eh Facil!";
	int cont_mai, cont_min;
	strcontMaiMin(frase,&cont_mai, &cont_min);
	printf("%d %d",cont_mai, cont_min);
}
