Switch - Break


int main(){
	int a = 0;
	switch(a){
		case(0): printf("Olá ");
		case(1): printf(" Mundo ");
		case(2): printf("!"); break;
		case(3): printf(" Tchau"); break;
		default: printf("Erro, valor errado!\n");
	}
}

For e Vetores


const int tamanho = 20;
int main(){
	int i, V[tamanho] = {0};
	int soma = 0;
	// valores aleatorios no vetor
	for(i=0;i < tamanho; i++)
		V[i] = rand() % tamanho; // Valor aleatorio de 0 a tamanho -1
	// Faz a soma dos valores
	for(i=0;i < tamanho; i++)
		soma += V[i];
	printf("%d\n",soma);
}

While e Do-While


int main(){
	int valor;
	do{
		printf("Digite um valor, 0 para sair");
		scanf("%d",&valor);
		printf("O valor digitado foi %d",valor);
	} while(valor != 0);
}
